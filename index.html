const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); const scoreEl = document.getElementById('score'); const bestEl = document.getElementById('best'); const GRID = 8; let grid = Array(GRID).fill().map(() => Array(GRID).fill(null)); let score = 0; let bestScore = localStorage.getItem('bb_best_v2') || 0; let shapes = []; let drag = null; // Координаты цели и текущие координаты (для плавности) let targetX = 0, targetY = 0; let curX = 0, curY = 0; let cellSize, boardSize, boardX, boardY, dockY, dockHeight; const COLORS = [ { m: '#FF5E57', s: '#D63031' }, { m: '#1DD1A1', s: '#10AC84' }, { m: '#54A0FF', s: '#2E86DE' }, { m: '#FECA57', s: '#FF9F43' }, { m: '#5F27CD', s: '#341F97' }, { m: '#00d2d3', s: '#01a3a4' } ]; function getRandomShape() { const sets = [ [[0,0],[1,0],[2,0],[0,1],[1,1],[2,1],[0,2],[1,2],[2,2]], // 3x3 [[0,0],[1,0],[2,0],[0,1],[1,1],[2,1]], // 3x2 [[0,0],[1,0],[0,1],[1,1]], // 2x2 [[0,0],[0,1],[1,1]], // L [[0,0],[1,0],[2,0]], // 3x1 [[0,0],[0,1],[0,2]], // 1x3 [[0,0],[1,0],[1,1],[2,1]], // Z [[0,0]], // 1x1 [[0,0],[1,0]] // 2x1 ]; return sets[Math.floor(Math.random() * sets.length)]; } function resize() { const dpr = window.devicePixelRatio || 1; const rect = canvas.parentElement.getBoundingClientRect(); canvas.width = rect.width * dpr; canvas.height = rect.height * dpr; ctx.scale(dpr, dpr); boardSize = rect.width * 0.94; cellSize = boardSize / GRID; boardX = (rect.width - boardSize) / 2; boardY = 10; dockY = boardY + boardSize + 40; dockHeight = rect.height - dockY; bestEl.innerText = `РЕКОРД: ${bestScore}`; } function spawnShapes() { shapes = []; for (let i = 0; i < 3; i++) { shapes.push({ blocks: getRandomShape(), color: COLORS[Math.floor(Math.random() * COLORS.length)], active: true, slot: i }); } if (!canMoveAny()) gameOver(); } function canFit(shape, gx, gy) { return shape.blocks.every(b => { let nx = gx + b[0], ny = gy + b[1]; return nx >= 0 && nx < GRID && ny >= 0 && ny < GRID && !grid[ny][nx]; }); } function canMoveAny() { return shapes.some(s => { if (!s.active) return false; for (let r = 0; r < GRID; r++) { for (let c = 0; c < GRID; c++) { if (canFit(s, c, r)) return true; } } return false; }); } function drawBlock(x, y, color, size, alpha = 1) { ctx.save(); ctx.globalAlpha = alpha; ctx.fillStyle = color.m; ctx.beginPath(); const r = size * 0.2; ctx.roundRect(x + 1.5, y + 1.5, size - 3, size - 3, r); ctx.fill(); ctx.fillStyle = 'rgba(255,255,255,0.15)'; ctx.fillRect(x + size*0.2, y + size*0.15, size*0.6, size*0.15); ctx.restore(); } function render() { const rect = canvas.parentElement.getBoundingClientRect(); ctx.clearRect(0, 0, rect.width, rect.height); // Сетка for (let r = 0; r < GRID; r++) { for (let c = 0; c < GRID; c++) { ctx.fillStyle = '#f0f0f0'; ctx.beginPath(); ctx.roundRect(boardX + c*cellSize + 1, boardY + r*cellSize + 1, cellSize - 2, cellSize - 2, 6); ctx.fill(); if (grid[r][c]) drawBlock(boardX + c*cellSize, boardY + r*cellSize, grid[r][c], cellSize); } } // Фигуры в доке const slotW = rect.width / 3; shapes.forEach(s => { if (!s.active || s === drag) return; const miniSize = cellSize * 0.7; const sW = Math.max(...s.blocks.map(b => b[0])) * miniSize; const sH = Math.max(...s.blocks.map(b => b[1])) * miniSize; const x = s.slot * slotW + (slotW - sW) / 2 - miniSize/2; const y = dockY + (dockHeight/2 - sH/2) - miniSize/2; s.blocks.forEach(b => drawBlock(x + b[0]*miniSize, y + b[1]*miniSize, s.color, miniSize)); }); // Перетаскивание if (drag) { // Плавное следование (Lerp) curX += (targetX - curX) * 0.25; curY += (targetY - curY) * 0.25; const gx = Math.round((curX - boardX - cellSize/2) / cellSize); const gy = Math.round((curY - boardY - cellSize/2) / cellSize); // Превью на сетке if (canFit(drag, gx, gy)) { drag.blocks.forEach(b => { drawBlock(boardX + (gx+b[0])*cellSize, boardY + (gy+b[1])*cellSize, drag.color, cellSize, 0.3); }); } // Сама фигура (приподнята над пальцем на 70px для удобства) drag.blocks.forEach(b => { const drawX = curX + (b[0] - 0.5) * cellSize; const drawY = curY + (b[1] - 1.5) * cellSize - 70; drawBlock(drawX, drawY, drag.color, cellSize); }); } requestAnimationFrame(render); } function handleEnd() { if (!drag) return; // При расчете места учитываем смещение -70px const gx = Math.round((curX - boardX - cellSize/2) / cellSize); const gy = Math.round((curY - 70 - boardY - cellSize/2) / cellSize); if (canFit(drag, gx, gy)) { drag.blocks.forEach(b => grid[gy+b[1]][gx+b[0]] = drag.color); drag.active = false; let fullRows = [], fullCols = []; for(let r=0; r v)) fullRows.push(r); for(let c=0; c r[c])) fullCols.push(c); const lines = fullRows.length + fullCols.length; if (lines > 0) { // КОМБО СИСТЕМА: бонус растет экспоненциально // 1 линия = 100, 2 = 300, 3 = 600, 4 = 1000 и т.д. const comboBonus = (lines * (lines + 1) / 2) * 100; score += comboBonus; // Визуальный эффект scoreEl.style.transform = 'scale(1.2)'; setTimeout(() => scoreEl.style.transform = 'scale(1)', 200); scoreEl.innerText = score; fullRows.forEach(r => grid[r].fill(null)); fullCols.forEach(c => { for(let r=0; r !s.active)) spawnShapes(); else if (!canMoveAny()) gameOver(); } drag = null; } function gameOver() { document.getElementById('overlay').style.display = 'flex'; document.getElementById('final-score').innerText = `СЧЕТ: ${score}`; if (score > bestScore) localStorage.setItem('bb_best_v2', score); } function updateMousePos(e) { const rect = canvas.getBoundingClientRect(); const clientX = e.touches ? e.touches[0].clientX : e.clientX; const clientY = e.touches ? e.touches[0].clientY : e.clientY; targetX = clientX - rect.left; targetY = clientY - rect.top; } canvas.addEventListener('touchstart', e => { updateMousePos(e); curX = targetX; curY = targetY; // Мгновенно перемещаем к началу касания const slotW = canvas.offsetWidth / 3; if (targetY > dockY - 40) { const slotIdx = Math.floor(targetX / slotW); const s = shapes.find(sh => sh.slot === slotIdx && sh.active); if (s) drag = s; } }); window.addEventListener('touchmove', e => { if(drag) { updateMousePos(e); e.preventDefault(); } }, {passive: false}); window.addEventListener('touchend', handleEnd); canvas.addEventListener('mousedown', e => { updateMousePos(e); curX = targetX; curY = targetY; const slotW = canvas.offsetWidth / 3; if (targetY > dockY - 40) { const slotIdx = Math.floor(targetX / slotW); const s = shapes.find(sh => sh.slot === slotIdx && sh.active); if (s) drag = s; } }); window.addEventListener('mousemove', e => { if(drag) updateMousePos(e); }); window.addEventListener('mouseup', handleEnd); window.addEventListener('resize', resize); resize(); spawnShapes(); render();
